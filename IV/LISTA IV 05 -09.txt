import random

lista = []
maior = 0
menor = 100

for i in range(10):
	n = random.randint(1,100)
	lista.append(n)

	if n < menor:
		menor = n
	if n > maior:
		maior = n

for x in lista:
    print(x)

exercicio 2

import random

lista = []
par = []
impar = []

for i in range(20):
    n = random.randint(1,100)
    lista.append(n)
    if n % 2 == 0:
        par.append(n)
    else:
        impar.append(n)

print ('Numeros sorteados')
for x in lista:
    print(x)
print('PAR')
for x in par:
    print(x)
print('IMPAR')    
for x in impar:
    print(x)
print('LISTA =', lista)
print('PAR = ', par)
print('IMPAR =', impar)
